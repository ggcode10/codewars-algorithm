# 최대공약수 - 1850

```swift
import Foundation

// 최대공약수
func gcd(_ a: Int, _ b: Int) -> Int {
    if a % b == 0 {
        return b
    }
    return gcd(b, a % b)
}

var input = readLine()!.split(separator: " ").map { Int($0)! }
var A = input[0], B = input[1]

let result = gcd(A, B)

for _ in 0..<result {
    print("1", terminator: "")
}

```
