# 최소공배수 - 13241

```swift
import Foundation

func gcd(_ a: Int, _ b: Int) -> Int {
    if b == 0 {
        return a
    }
    return gcd(b, a % b)
}

let input = readLine()!.split(separator: " ").map { Int($0)! }
var A = input[0], B = input[1]
print(A * B / gcd(A, B))
```
