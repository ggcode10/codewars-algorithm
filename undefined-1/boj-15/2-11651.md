# 좌표 정렬하기 2 - 11651

```swift
import Foundation

var N = Int(readLine()!)!
var X: [(Int, Int)] = []
for _ in 0..<N {
    let input = readLine()!.split(separator: " ").map{ Int($0)! }
    let x = input[0], y = input[1]
    X.append((x, y))
}

X.sort(by: {
    $0.1 == $1.1 ? $0.0 < $1.0 : $0.1 < $1.1
})

for i in 0..<N {
    print("\(X[i].0) \(X[i].1)")
}

```
