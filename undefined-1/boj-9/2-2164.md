```swift
import Foundation

var N = Int(readLine()!)!
var queue: [Int] = Array(1...N)
var tmp: Int = 0

if N == 1 {
    print(1)
} else {
    while true {
        queue[tmp] = 0
        queue.append(queue[tmp + 1])
        queue[tmp + 1] = 0
        
        if queue[queue.count - 2] == 0 {
            print(queue.last!)
            break
        }
        tmp += 2
    }
}
```
