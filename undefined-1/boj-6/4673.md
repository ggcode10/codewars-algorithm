# 셀프 넘버 - 4673번

```swift
import Foundation

var s: Set<Int> = []
for i in 1...10000 {
    s.insert(selfNum(i))
}

for j in 1...10000 {
    if !s.contains(j) { print(j) }
}

func selfNum(_ n: Int) -> Int {
    var sum = n, now = n
    while now != 0 {
        sum += now % 10
        now /= 10
        
        if n == 0 { break }
    }
    return sum
}
```
