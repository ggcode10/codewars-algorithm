# 삼각형과 세 변 - 5073

```swift
import Foundation

while true {
    let input = readLine()!.split(separator: " ").map { Int($0)! }
    let a = input[0]
    let b = input[1]
    let c = input[2]
    
    if a == 0 && b == 0 && c == 0 {
        break
    }
    
    let sides = [a, b, c].sorted()
    
    if sides[0] + sides[1] <= sides[2] {
        print("Invalid")
    } else if sides[0] == sides[2] {
        print("Equilateral")
    } else if sides[0] == sides[1] || sides[1] == sides[2] {
        print("Isosceles")
    } else {
        print("Scalene")
    }
}

```
