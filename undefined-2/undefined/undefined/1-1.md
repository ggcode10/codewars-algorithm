# 수 조작하기 1

```swift
import Foundation

func solution(_ n:Int, _ control:String) -> Int {
    var j = n
    for i in control {
        if i == "w" {
            j += 1
        } else if i == "s" {
            j -= 1
        } else if i == "d" {
            j += 10
        } else if i == "a" {
            j -= 10
        }
    }
    return j
}
```
