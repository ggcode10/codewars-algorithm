# 문자열 반복 - 2675

```swift
import Foundation

// 총 개수
var input = Int(readLine()!)!

for _ in 1...input {
    // 반복돌릴 글자
    let input = readLine()!.split(separator: " ")
    let R = Int(input[0])!, str = input[1]
    
    for j in str {
        for _ in 1...R {
            print("\(j)", terminator: "")
        }
    }
    print(" ")
}
```
