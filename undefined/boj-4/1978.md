# 소수 찾기 - 1978

```swift
// Some code
var N = Int(readLine()!)!
var input = readLine()!.split(separator: " ").map{ Int($0)! }
var count = 0

for i in input {
    
    if i == 1 {
        count += 1
        continue
    }
    
    for j in 2..<i {
        if i % j == 0 {
            count += 1
            break
        }
    }
}
print(input.count - count)
```
