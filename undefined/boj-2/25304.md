# 영수증 - 25304

```swift
import Foundation

// 총 금액 X
// 구매한 물건 수 N

var X = Int(readLine()!)! // 총 금액
var N = Int(readLine()!)! // 물건 수

var sum = 0

(1...N).forEach{ _ in
    let input = readLine()!.split(separator: " ").map{ Int($0)! }
    let A = input[0], B = input[1] // 가격과 갯수 곱하기
    let result = A * B
    sum += result
}

print(X == sum ? "Yes" : "No")

```
