# 보물 - 1026

```swift
// Some code
import Foundation
// A의 가장 작은 수와  B의 가장 큰수를 곱해
// 반복문으로 해야겠네

var N = Int(readLine()!)!
var A = readLine()!.split(separator: " ").map{ Int($0)! }
var B = readLine()!.split(separator: " ").map{ Int($0)! }

var sortA = A.sorted(by: >)
var sortB = B.sorted(by: <)
var sum = 0

for _ in 0..<N {
    sum += sortA.removeLast() * sortB.removeLast()
}
print(sum)

```
