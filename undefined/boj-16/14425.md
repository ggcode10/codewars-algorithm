# 문자열 집합 - 14425

```
import Foundation

var input = readLine()!.split(separator: " ").map{ Int($0)! }
var `set`: Set<String> = []
var answer = 0
var N = input[0], M = input[1]

for _ in 0..<N {
    set.insert(readLine()!)
}

for _ in 0..<M {
    let word = readLine()!
    answer += set.contains(word) ? 1: 0
}
print(answer)

```
