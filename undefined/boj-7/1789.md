# 수들의 합- 1789

```swift
import Foundation

var S = Int(readLine()!)!
/*
 더해서 S이되는 숫자들..
 */
var sum = 0
var i = 1
var answer = 0
while true {
    sum += i
    answer += 1
    i += 1
    
    if sum == S { break }
    else if sum > S {
        answer -= 1
        break
    }
}
print(answer)

```
