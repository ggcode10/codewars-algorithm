# 단어순서 뒤집기 - 12605

```swift
import Foundation

let n = Int(readLine()!)!

for i in 1...n {
    let input = readLine()!.split(separator: " ")
    let reversedWords = input.reversed()
    let result = reversedWords.joined(separator: " ")
    
    print("Case #\(i): \(result)")
}
```
