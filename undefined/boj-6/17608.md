# 막대기 - 17608

```swift
import Foundation

var N = Int(readLine()!)!
var stack: [Int] = []

for _ in 0..<N {
    var bar: Int = Int(readLine()!)!
    // 스택이 비어있지 않거나, 스택의 마지막 막대기가 지금 들어온 것 보다 작으면
    while !stack.isEmpty, stack.last! <= bar {
        stack.removeLast()
    }
    stack.append(bar)
}

print(stack.count)

```
