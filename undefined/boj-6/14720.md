# 우유 축제 - 14720

```swift
import Foundation

// 0 딸기
// 1 초코
// 2 바나나

var N = Int(readLine()!)!

var input = readLine()!.split(separator: " ").map{ Int($0)! }
let milk = [0,1,2]
var j: Int = 0
var count = 0

for i in 0..<N {
    if input[i] == milk[j] {
        if j == 2 {
            j = 0
        } else {
            j += 1
        }
        count += 1
    }
}
print(count)

```
