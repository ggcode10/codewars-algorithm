```swift
let input = readLine()!.split(separator: " ").map{Int($0)!}
var arr: [String] = []
for _ in 0..<input[0]{
    arr.append(readLine()!)
}
arr.sort()
var count = 0
var res: [String] = []
for _ in 0..<input[1]{
    let read = readLine()!
    if binarySearch(array: arr, item: read){
        count += 1
        res.append(read)
    }
}
print(count)
res.sort()
res.forEach{print($0)}

func binarySearch(array: [String], item: String) -> Bool {
    var low = 0
    var high = array.count - 1
    
    while low <= high {
        let mid = (low + high) / 2
        let guess = array[mid]
        if guess == item {
            return true
        } else if guess > item {
            high = mid - 1
        } else {
            low = mid + 1
        }
    }
    
    return false
}
```
