# 단어 뒤집기 - 9093

```swift
var T = Int(readLine()!)!

var strArr: [String] = []

for _ in 1...T {
    strArr.append(readLine()!)
}

for i in strArr {
    let string = i.split(separator: " ")
    var reverseString = ""
    for j in 0..<string.count {
        reverseString += string[j].reversed() + " "
    }
    print(reverseString)
}
```
