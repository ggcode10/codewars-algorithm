# 알람 시계 - 2884

```swift
import Foundation

var input = readLine()!.split(separator: " ").map { Int($0)! }
var H = input[0], M = input[1]

if M < 45 && M != 45 {
    if H == 0 {
        H = 23
    } else {
        H = H - 1
    }
    
    M = M + 15
} else {
    M = M - 45
}
print(H, M)
```
