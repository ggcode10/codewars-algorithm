# 대지 - 9063

```swift
import Foundation
var n = Int(readLine()!)!
var x: [Int] = []
var y: [Int] = []

for _ in 0..<n {
    let input = readLine()!.split(separator: " ").map{ Int($0)! }
    let A = input[0], B = input[1]
    x.append(A)
    y.append(B)
}
var X = x.max()! - x.min()!
var Y = y.max()! - y.min()!
print(X * Y)
```
