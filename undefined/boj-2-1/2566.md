# 최댓값 - 2566

```swift
import Foundation

// 2차원 배열
var arr:[[Int]] = []

for _ in 1...9 {
    arr.append(readLine()!.split(separator: " ").compactMap{ Int($0) })
}

var maxValue = arr.flatMap{ $0 }.max()!

for y in 1..<9 {
    for x in 1..<9 {
        if arr[y][x] == maxValue {
            print(maxValue)
            print(y + 1, x + 1)
            break
        }
    }
}
```
