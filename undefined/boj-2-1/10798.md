# 세로 읽기 - 10798

```swift
import Foundation

// 15개 글자 배열 5개 생성
var words = [[String]](repeating: [String](repeating: "", count: 15), count: 5)
// 5줄 입력
for i in 0..<5 {
    var input = readLine()!.map{ String($0) }
    var lastIndex = input.count - 1
    words[i].replaceSubrange(0..<lastIndex, with: input)
}

for j in 0..<15 {
    for x in 0..<5 {
        print(words[x][j], terminator: "")
    }
}

```
